

import:--import
	declare @t_date nvarchar(10) = [1]
	declare @t_driverno nvarchar(20) = [2]
	
	declare @tmp table(
		carno nvarchar(20)
		,[money] float
		,tablea nvarchar(20)
		,accy nvarchar(20)
		,noa nvarchar(20)
		,datea nvarchar(20)
	)
	insert into @tmp(carno,[money],tablea,accy,noa,datea)
	select carno,isnull([money],0) [money],'vcc',accy,noa,datea
	from view_vcc 
	where datea=@t_date and driverno=@t_driverno
	union all
	select carno,isnull([money],0) [money],'rc2',accy,noa,datea 
	from view_rc2
	where datea=@t_date and driverno=@t_driverno
	select * from @tmp order by datea desc;
