into @noa,@paytype

begin
	set @num = ''
	set @i = 1

	begin
		if(ASCII(SUBSTRING(@paytype,@i,1))>=48 and ASCII(SUBSTRING(@paytype,@i,1))<=57)

	into @noa,@paytype

					select @totmount=@totmount+@mount,@totweight=@totweight+@weight
				end
				else

					select @totmount=@mount,@totweight=@weight
					set @date=@curdate
